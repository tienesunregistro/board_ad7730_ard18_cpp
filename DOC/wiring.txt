MSBFIRST
Ejemplos	
int data = 0; // Pin de wiring 0 para data 
int clock = 1;  // Pin de wiring 1 para clock 
int strobe = 2;  // Pin de wiring 2 para strobe (latch) 
byte counter = 0;

void setup() {
  pinMode(data OUTPUT);
  pinMode(clock, OUTPUT);
  pinMode(strobe, OUTPUT);
}

void loop() { 
  digitalWrite(strobe, LOW);
  shiftOut(data, clock, MSBFIRST, counter);  // escribe counter en el registrador  
  digitalWrite(strobe, HIGH); 
  delay(1000);
  counter = counter + 1;
}
